uint64_t fibo_expo_var(uint64_t n)
{
    uint64_t a_t = 1, b_t = 0, d_t = 1, // "1"
            a_y = 0, b_y = 1, d_y = 1; // "y"

    while (n)
    {
        if (n & 1)
        {
            // t*=y;
            uint64_t bx = b_t * b_y;
            b_t = a_t * b_y + b_t * d_y;
            a_t = a_t * a_y + bx;
            d_t = bx + d_t * d_y;
        }

        //y*=y;
        uint64_t b2 = b_y * b_y;
        b_y *= (a_y + d_y);
        a_y = a_y * a_y + b2;
        d_y = b2 + d_y * d_y;

        n >>= 1;
    }
    return b_t;
}
